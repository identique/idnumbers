 57def checksum(id_number) -> Optional[CHECK_DIGIT]:
 58    """
 59    check the checksum
 60    https://www.oecd.org/tax/automatic-exchange/crs-implementation-and-assistance/tax-identification-numbers/Kazakhstan-TIN.pdf
 61    """
 62    numbers = [int(char) for char in id_number]
 63    modulus = weighted_modulus_digit(numbers[0:-1], WEIGHTS1, 11, True)
 64    if modulus == 10:
 65        modulus = weighted_modulus_digit(numbers[0:-1], WEIGHTS2, 11, True)
 66        # the second modulus will not be 10. If it is, it's wrong id number
 67    return modulus if modulus < 10 else None

 70class IndividualIDNumber:
 71    """
 72    Kazakhstan individual identification number, ЖСН, ZhSN, ИИН, IIN
 73    https://en.wikipedia.org/wiki/National_identification_number#Kazakhstan
 74    https://korgan-zan.kz/en/obtaining-iin-and-bin-in-kazakhstan/
 75    https://www.oecd.org/tax/automatic-exchange/crs-implementation-and-assistance/tax-identification-numbers/Kazakhstan-TIN.pdf
 76    """
 77    METADATA = SimpleNamespace(**{
 78        'iso3166_alpha2': 'KZ',
 79        'min_length': 12,
 80        'max_length': 12,
 81        'parsable': True,
 82        'checksum': True,
 83        'regexp': re.compile(r'^(?P<yy>\d{2})(?P<mm>\d{2})(?P<dd>\d{2})'
 84                             r'(?P<century>\d)'
 85                             r'(?P<sn>\d{4})'
 86                             r'(?P<checksum>\d)$')
 87    })
 88
 89    @staticmethod
 90    def validate(id_number: str) -> bool:
 91        """validate"""
 92        if not validate_regexp(id_number, IndividualIDNumber.METADATA.regexp):
 93            return False
 94        return IndividualIDNumber.parse(id_number) is not None
 95
 96    @staticmethod
 97    def parse(id_number: str) -> Optional[IINParseResult]:
 98        """parse the result"""
 99        match_obj = IndividualIDNumber.METADATA.regexp.match(id_number)
100        if not match_obj:
101            return None
102        iin_checksum = IndividualIDNumber.checksum(id_number)
103        if iin_checksum is None or iin_checksum != int(match_obj.group('checksum')):
104            return None
105        gender_year_base = IndividualIDNumber.get_gender_year_base(int(match_obj.group('century')))
106        if not gender_year_base:
107            return None
108        gender, year_base = gender_year_base
109        yy = int(match_obj.group('yy'))
110        mm = int(match_obj.group('mm'))
111        dd = int(match_obj.group('dd'))
112        try:
113            return {
114                'yyyymmdd': date(yy + year_base, mm, dd),
115                'gender': gender,
116                'sn': match_obj.group('sn'),
117                'checksum': int(match_obj.group('checksum')),
118            }
119        except ValueError:
120            # catch the date error
121            return None
122
123    @staticmethod
124    def checksum(id_number) -> Optional[CHECK_DIGIT]:
125        if not validate_regexp(id_number, IndividualIDNumber.METADATA.regexp):
126            return False
127        return checksum(id_number)
128
129    @staticmethod
130    def get_gender_year_base(century: CHECK_DIGIT) -> Optional[Tuple[Gender, int]]:
131        gender = Gender.MALE if century % 2 == 1 else Gender.FEMALE
132        if 1 <= century < 3:
133            year_base = 1800
134        elif 3 <= century < 5:
135            year_base = 1900
136        elif 5 <= century < 7:
137            year_base = 2000
138        else:
139            return None
140        return gender, year_base
